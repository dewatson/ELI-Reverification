# Assign the decision point
DecisionPoint <- 2

# Select Armbruster 'normal' samples
D_norm <- subset(lowCOdata, Candidate == FALSE, select = Response)
# Select Armbruster 'carryover candidate' samples
D_cc <- subset(lowCOdata, Candidate == TRUE, select = Response)
# Calculate the carryover
cat("\n", "Carryover (%): ", sep = "")

## 
## Carryover (%):

write(carryover(D_cc$Response, D_norm$Response), "")

## 0


# Calculate the CV of the normal samples
cat("\n", "CV of normal decision point samples: ", sep = "")

## 
## CV of normal decision point samples:

write(round(100 * sd(D_norm$Response)/mean(D_norm$Response), 1), "")

## 2.8


# Unpaired, two-sample t.test using unequal variance (Welch): check whether
# the difference in means is <0.4 (20% of the decision level)
t.test(D_norm$Response, D_cc$Response, mu = 0.2 * DecisionPoint, alternative = "less")

## 
##  Welch Two Sample t-test
## 
## data:  D_norm$Response and D_cc$Response

## alternative hypothesis: true difference in means is less than 0.4

##     -Inf 0.06442
## sample estimates:

# Read in cross-reactivity data
crossrx <- read.csv("data/cRCR.csv", header = TRUE)

# Calculate ascorbic acid cross-reactivity
ascorbicacid <- subset(crossrx, Analyte == "ascorbate", select = c(Concentration, 
    Response))
ascorbicacid$Concentration <- ascorbicacid$Concentration * 100
attach(ascorbicacid)
aa_fit <- lm(Response ~ Concentration)
aa_summary <- summary(aa_fit)
aa_cross_reactivity <- 100 * 20 * coef(aa_fit)[2]/(20 - coef(aa_fit)[1])
cat("\n", "Ascorbic acid cross-reactivity (%): ", sep = "")

## 
## Ascorbic acid cross-reactivity (%):

write(round(as.numeric(aa_cross_reactivity), 1), "")

## 0.3

## 
## Ascorbic acid linear regression:

aa_fit$coefficients
